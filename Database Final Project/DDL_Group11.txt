--CODE MUST BE RAN IN TWO PARTS STARTING FROM HERE:

CREATE TABLE USER_ACCOUNT (  
USER_ID INTEGER NOT NULL,  
USERNAME VARCHAR(35) UNIQUE NOT NULL,  
PASSWORD VARCHAR(35) UNIQUE NOT NULL CHECK (LENGTH(PASSWORD) >= 8 AND INSTR(PASSWORD, '$') > 0 OR INSTR(PASSWORD, '!')  > 0 OR INSTR(PASSWORD, '@') > 0 
    OR INSTR(PASSWORD, '#') > 0 OR INSTR(PASSWORD, '%') > 0 OR INSTR(PASSWORD, '^') > 0 OR INSTR(PASSWORD, '&') > 0 OR INSTR(PASSWORD, '*') > 0 
    OR INSTR(PASSWORD, '(') > 0 OR INSTR(PASSWORD, ')') > 0 OR INSTR(PASSWORD, '-') > 0 OR INSTR(PASSWORD, '=') > 0 OR INSTR(PASSWORD, '_') > 0 
    OR INSTR(PASSWORD, '+') > 0 OR INSTR(PASSWORD, '`') > 0 OR INSTR(PASSWORD, '~') > 0 OR INSTR(PASSWORD, '[') > 0 OR INSTR(PASSWORD, ']') > 0 
    OR INSTR(PASSWORD, '{') > 0 OR INSTR(PASSWORD, ';') > 0 OR INSTR(PASSWORD, ':') > 0 OR INSTR(PASSWORD, ',') > 0 OR INSTR(PASSWORD, '<') > 0 
    OR INSTR(PASSWORD, '.') > 0 OR INSTR(PASSWORD, '/') > 0 OR INSTR(PASSWORD, '\') > 0 OR INSTR(PASSWORD, '|') > 0), 
F_NAME CHAR(35) NOT NULL CHECK (F_NAME NOT LIKE '%[^A-Z]%'),  
L_NAME CHAR(35) NOT NULL CHECK (L_NAME NOT LIKE '%[^A-Z]%'),  
CREATION_DATE DATE NOT NULL,  
DISABILITY SMALLINT NOT NULL CHECK(DISABILITY IN (0,1)),  
AGE SMALLINT NOT NULL CHECK(AGE >=18),  
PRIMARY KEY (USER_ID)  
);  

CREATE TABLE PASS (  
PASS_ID INTEGER UNIQUE NOT NULL,  
USER_ID INTEGER UNIQUE NOT NULL,  
THIRTY_DAY SMALLINT NOT NULL CHECK(THIRTY_DAY IN (0,1)),  
SEVEN_DAY SMALLINT NOT NULL CHECK(SEVEN_DAY IN (0,1)),  
ALL_DAY SMALLINT NOT NULL CHECK(ALL_DAY IN (0,1)),  
PRICE SMALLINT NOT NULL CHECK(PRICE IN (5, 15, 50)),  
PRIMARY KEY (PASS_ID, USER_ID),  
FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT(USER_ID)  
);  

CREATE TABLE TICKET (  
TIX_ID INTEGER UNIQUE NOT NULL,  
USER_ID INTEGER UNIQUE,  
SINGLE SMALLINT NOT NULL CHECK(SINGLE IN (0,1)),  
DISCOUNT SMALLINT NOT NULL CHECK(DISCOUNT IN (0,1)), 
TRANSFER SMALLINT NOT NULL CHECK(TRANSFER IN (0,1)), 
PRICE SMALLINT NOT NULL CHECK(PRICE IN (2,1,1)),  
PRIMARY KEY (TIX_ID, USER_ID),  
FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT(USER_ID)  
);  

CREATE TABLE CITY (  
CITY_ID INTEGER NOT NULL,  
NAME VARCHAR(35) NOT NULL CHECK(NAME IN ('Orlando', 'Tampa', 'Lakeland')),  
ZIP CHAR(5) NOT NULL CHECK(LENGTH(ZIP) = 5),  
COUNTY VARCHAR(35) NOT NULL CHECK(COUNTY NOT LIKE '%[^A-Z]%'),  
PRIMARY KEY (CITY_ID)   
);  

CREATE TABLE ROUTES (  
ROUTE_ID INTEGER NOT NULL,  
NUM_STOPS INTEGER NOT NULL, 
DISTANCE INTEGER NOT NULL,  
DIRECTION VARCHAR(35) NOT NULL CHECK(DIRECTION IN ('Northbound','Southbound','Westbound','Eastbound')),   
PRIMARY KEY (ROUTE_ID)  
);

CREATE TABLE BUS (  
BUS_ID INTEGER NOT NULL,  
DRIVER_ID INTEGER NOT NULL,  
SERVICE_DATE DATE NOT NULL,  
MILEAGE INTEGER NOT NULL,  
CITY_ID INTEGER NOT NULL,  
ROUTE_ID INTEGER NOT NULL,  
PRIMARY KEY (BUS_ID),  
FOREIGN KEY (CITY_ID) REFERENCES CITY(CITY_ID),  
FOREIGN KEY (ROUTE_ID) REFERENCES ROUTES(ROUTE_ID)  
);  

CREATE TABLE STOPS (  
STOP_ID INTEGER NOT NULL,  
BUS_ID INTEGER NOT NULL,  
ROUTE_ID INTEGER NOT NULL,   
LATITUDE INTEGER NOT NULL CHECK(LATITUDE >= -90 AND LATITUDE <= 90),  
LONGITUDE INTEGER NOT NULL CHECK(LONGITUDE >= -180 AND LONGITUDE <= 180),  
STREET_NAME VARCHAR(35) NOT NULL CHECK(STREET_NAME NOT LIKE '%[^A-Z]%'),  
Arrival_Time TIMESTAMP NOT NULL, 
Departure_Time TIMESTAMP NOT NULL, 
PRIMARY KEY (STOP_ID),  
FOREIGN KEY (BUS_ID) REFERENCES BUS(BUS_ID),  
FOREIGN KEY (ROUTE_ID) REFERENCES ROUTES(ROUTE_ID)  
);  

CREATE TABLE PASSENGER (  
PASSENGER_ID INTEGER NOT NULL,  
USE_TIME DATE NOT NULL,  
TIX_ID INTEGER UNIQUE,   
PASS_ID INTEGER,  
USER_ID INTEGER, 
BUS_ID INTEGER NOT NULL,  
PRIMARY KEY (PASSENGER_ID),  
FOREIGN KEY (TIX_ID) REFERENCES TICKET(TIX_ID),  
FOREIGN KEY (PASS_ID) REFERENCES PASS(PASS_ID),  
FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT(USER_ID),  
FOREIGN KEY (BUS_ID) REFERENCES BUS(BUS_ID)  
);
